    enum Config¶

    enum zmbt::lang::Operator::Config {
    Null,
    Decor = 1U << 0,
    Comparison = 1U << 1,
    Arithmetics = 1U << 2,
    Bitwise = 1U << 3,
    Shift = 1U << 4,
    Logic = 1U << 5,
    Default = Decor|Comparison|Arithmetics|Bitwise|Shift,
    Full = Default|Logic
};
