139 auto get_next_ts = [](FB& fb) -> std::size_t {
140 std::size_t ts = std::string::npos;
141 if (fb.args) {
142 auto& dq = *std::static_pointer_cast<std::deque<Frame<ArgsTuple>>>(fb.args);
143 if (!dq.empty()) ts = std::min(ts, dq.front().ts);
144 }
145 if (fb.ret) {

147 if (!dq.empty()) ts = std::min(ts, dq.front().ts);

149 if (fb.err) {

151 if (!dq.empty()) ts = std::min(ts, dq.front().ts);
152 }
153 return ts;

156 struct Cursor {
157 std::size_t ts {std::string::npos}; // next timestamp
158 ThreadFrameBuffers* buffs{nullptr};
159 };
160
161 // initialize candidates
162 std::vector<Cursor> cursors;
163 cursors.reserve(fbs.size());
164 for (auto& tfbs : fbs) {
165 cursors.push_back({get_next_ts(tfbs.second), &tfbs});
166 }
167
168 Cursor null{};
169 for (;;) {
170 // find candidate with smallest ts
171 Cursor* cursor = &null;
172
173 for (auto& candidate: cursors)
174 {
175 if (candidate.ts < cursor->ts) {
176 cursor = &candidate;
177 }
178 }
179 if (cursor == &null)
180 break; // all empty
181
182 boost::json::object json_frame{

184 };
185
186 FB& fb = cursor->buffs->second;
187
188 if (tid_enabled)

190 json_frame["tid"] = tid2str(cursor->buffs->first);
191 }
192
193 // consume exactly one element of each deque matching this ts
194 if (fb.args) {
195 auto& dq = *std::static_pointer_cast<std::deque<Frame<ArgsTuple>>>(fb.args);
196 if (!dq.empty() && dq.front().ts == cursor->ts) {
197 json_frame["args"] = json_from(dq.front().data);
198 dq.pop_front();
199 }
200 }
201 if (fb.ret) {
202 auto& dq = *std::static_pointer_cast<std::deque<Frame<Return>>>(fb.ret);
203 if (!dq.empty() && dq.front().ts == cursor->ts) {
204 json_frame["return"] = json_from(dq.front().data);
205 dq.pop_front();
206 }
207 }
208 if (fb.err) {
209 auto& dq = *std::static_pointer_cast<std::deque<Frame<ErrorInfo>>>(fb.err);
210 if (!dq.empty() && dq.front().ts == cursor->ts) {
211 json_frame["exception"] = dq.front().data.to_json();
212 dq.pop_front();
213 }
