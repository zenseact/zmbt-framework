249 template <class T>
250 std::shared_ptr<T> GetShared(lang::Expression const& key_expr) const
251 {
252 boost::json::string const key = key_expr.eval().as_string();
253 auto found = data_->shared.cend();
254 {
255 auto lock = Lock();
256 found = data_->shared.find(key);
257 }
258 if (found == data_->shared.cend())

260 return nullptr;

266 throw environment_error("GetShared invoked with incompatible type for `%s`", key);

277 template <class T, class... A>
278 T& GetSharedRef(lang::Expression const& key_expr, A&&... args)
279 {
280 boost::json::string const key = key_expr.eval().as_string();
281 auto lock = Lock();
282 auto found = data_->shared.find(key);
283 if (found == data_->shared.cend())
284 {
285 auto const shared = std::make_shared<T>(std::forward<A>(args)...);
286 EnvironmentData::shared_data_record const record { typeid(T), shared };
287
288 if (not data_->shared.emplace(key, record).second)
289 {
290 throw environment_error("GetSharedRef failed to create shared object at `%s`", key);
291 }
292 return *shared;
293 }
294
295 auto const record = found->second;
296 if (std::type_index(typeid(T)) != record.first)
297 {
298 throw environment_error("GetSharedRef invoked with incompatible type for `%s`", key);
299 }
300
301 return *std::static_pointer_cast<T>(record.second);

392 Environment& RegisterTrigger(boost::json::string_view key, I&& interface, H&& host)
393 {
394 Trigger trigger{std::forward<H>(host), interface};
395 auto const json_data_key = format("/triggers/%s", key);
396 auto const err_msg = format("Trigger registering failed: key \"%s\" is taken", key);
397 auto lock = Lock();
398 auto const trigger_found = data_->triggers.find(key);

405 return *this;

409 RegisterInterface(key, interface, trigger.obj_id());

435 template <class I>
436 Environment& RegisterPrototypes(I&& interface)
437 {
438 static_assert(is_ifc_handle<I>::value, "");
439 auto const ifc_id = interface_id(interface);
440 auto const obj_id = object_id{ifc_host_nullptr<I>};
441 auto const proto_key = format("/prototypes/%s", ifc_id);
442 auto const defobj_key = format("/refs/defobj/%s", ifc_id);
443 auto const prototypes = reflect::prototypes<I>();
444
445 auto lock = Lock();
446 if (!data_->json_data.contains(proto_key))
447 {
448 data_->json_data(proto_key) = prototypes;
449 data_->json_data(defobj_key) = obj_id;

451 return *this;
