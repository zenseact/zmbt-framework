257 template <class T>
258 std::shared_ptr<T> GetShared(lang::Expression const& key_expr) const
259 {
260 boost::json::string const key = key_expr.eval().as_string();
261 auto found = data_->shared.cend();
262 {
263 auto lock = Lock();
264 found = data_->shared.find(key);
265 }
266 if (found == data_->shared.cend())
267 {
268 return nullptr;
269 }
270
271 auto const record = found->second;

273 {
274 throw_exception(environment_error("GetShared invoked with incompatible type for `%s`", key));
275 }
276

285 template <class T, class... A>
286 T& GetSharedRef(lang::Expression const& key_expr, A&&... args)
287 {
288 boost::json::string const key = key_expr.eval().as_string();
289 auto lock = Lock();
290 auto found = data_->shared.find(key);
291 if (found == data_->shared.cend())
292 {
293 auto const shared = std::make_shared<T>(std::forward<A>(args)...);
294 EnvironmentData::shared_data_record const record { typeid(T), shared };
295
296 if (not data_->shared.emplace(key, record).second)
297 {
298 throw_exception(environment_error("GetSharedRef failed to create shared object at `%s`", key));
299 }
300 return *shared;
301 }
302
303 auto const record = found->second;
304 if (std::type_index(typeid(T)) != record.first)
305 {
306 throw_exception(environment_error("GetSharedRef invoked with incompatible type for `%s`", key));
307 }
308
309 return *std::static_pointer_cast<T>(record.second);

408 {
409
410 interface_id const ifc_id{std::forward<Interface>(interface)};
411 object_id const obj_id{host};
412
413 auto recorder = GetRecorder(ifc_id, obj_id);
414
415 Trigger trigger{std::forward<H>(host), interface, recorder};
