211template <class T, std::size_t N>
212void dejsonize_array(boost::json::array const& jarr, T (&array)[N])
213{
214
215 if (jarr.size() != N)
216 {
217 throw serialization_error("JSON array size mismatch in deserialization of `%s[%d]`",
218 zmbt::type_name<T>() , N);
219 }
220 auto jarr_it = jarr.cbegin();
221 for (size_t i = 0; i < N; i++)
222 {
223 array[i] = zmbt::dejsonize<T>(*jarr_it++);
224 }
